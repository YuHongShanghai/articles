# 【从零手写播放器】第 12 章：播放器架构设计

> 前面我们分别实现了视频渲染和音频播放，但它们是独立的。一个真正的播放器需要同时处理音频和视频，并且保证两者同步。本章我们来设计播放器的整体架构，并实现核心的线程安全队列。

## 12.1 为什么需要多线程？

在单线程中，解封装、解码、渲染串行执行：

单线程方式串行执行（读包 → 视频解码 → 渲染 → 读包 → 音频解码 → 播放），问题严重：
- 视频解码可能耗时较长，音频播放会卡顿
- 音频回调需要在固定时间内返回数据，否则会出现杂音
- 无法充分利用多核 CPU

多线程方案将各环节并行化。

## 12.3 线程安全队列设计

播放器中有两种队列：

1. **PacketQueue**：存放 `AVPacket*`（压缩数据包）
2. **FrameQueue**：存放 `AVFrame*`（解码后的原始帧）

### 12.3.1 设计要求

- **线程安全**：多个线程同时读写
- **阻塞等待**：队列为空时读取方阻塞，队列满时写入方阻塞
- **可中止**：支持通知所有等待线程退出（用于播放器关闭、Seek 等场景）
- **有容量限制**：防止内存暴涨

### 12.3.2 PacketQueue 实现

```cpp
// chapter-12-queue/packet_queue.h

#pragma once

extern "C" {
#include <libavcodec/avcodec.h>
}

#include <queue>
#include <mutex>
#include <condition_variable>

class PacketQueue {
public:
    PacketQueue() = default;
    ~PacketQueue() { flush(); }

    // 禁止拷贝
    PacketQueue(const PacketQueue&) = delete;
    PacketQueue& operator=(const PacketQueue&) = delete;

    // 放入一个包（会创建引用，不会修改原始包）
    bool push(AVPacket* pkt) {
        std::unique_lock<std::mutex> lock(mutex_);

        // 等待队列未满或已中止
        cond_push_.wait(lock, [this] {
            return abort_ || size_ < max_size_;
        });

        if (abort_) return false;

        AVPacket* new_pkt = av_packet_alloc();
        av_packet_ref(new_pkt, pkt);
        queue_.push(new_pkt);
        size_++;
        total_duration_ += pkt->duration;

        cond_pop_.notify_one();
        return true;
    }

    // 取出一个包（阻塞等待）
    // 返回 true 表示成功取到，false 表示已中止
    bool pop(AVPacket* pkt) {
        std::unique_lock<std::mutex> lock(mutex_);

        cond_pop_.wait(lock, [this] {
            return abort_ || !queue_.empty();
        });

        if (abort_ && queue_.empty()) return false;

        AVPacket* front = queue_.front();
        queue_.pop();
        av_packet_move_ref(pkt, front);
        av_packet_free(&front);
        size_--;

        cond_push_.notify_one();
        return true;
    }

    // 清空队列
    void flush() {
        std::lock_guard<std::mutex> lock(mutex_);
        while (!queue_.empty()) {
            AVPacket* pkt = queue_.front();
            queue_.pop();
            av_packet_free(&pkt);
        }
        size_ = 0;
        total_duration_ = 0;
        cond_push_.notify_all();
    }

    // 中止：唤醒所有等待线程
    void abort() {
        std::lock_guard<std::mutex> lock(mutex_);
        abort_ = true;
        cond_pop_.notify_all();
        cond_push_.notify_all();
    }

    // 恢复
    void start() {
        std::lock_guard<std::mutex> lock(mutex_);
        abort_ = false;
    }

    int size() const {
        std::lock_guard<std::mutex> lock(mutex_);
        return size_;
    }

    void set_max_size(int max) {
        std::lock_guard<std::mutex> lock(mutex_);
        max_size_ = max;
        cond_push_.notify_all();
    }

private:
    std::queue<AVPacket*> queue_;
    mutable std::mutex mutex_;
    std::condition_variable cond_pop_;   // 通知消费者
    std::condition_variable cond_push_;  // 通知生产者
    int size_ = 0;
    int max_size_ = 128;                 // 默认最大 128 个包
    int64_t total_duration_ = 0;
    bool abort_ = false;
};
```

### 12.3.3 FrameQueue 实现

```cpp
// chapter-12-queue/frame_queue.h

#pragma once

extern "C" {
#include <libavutil/frame.h>
}

#include <queue>
#include <mutex>
#include <condition_variable>

class FrameQueue {
public:
    FrameQueue() = default;
    ~FrameQueue() { flush(); }

    FrameQueue(const FrameQueue&) = delete;
    FrameQueue& operator=(const FrameQueue&) = delete;

    // 放入一帧
    bool push(AVFrame* frame) {
        std::unique_lock<std::mutex> lock(mutex_);

        cond_push_.wait(lock, [this] {
            return abort_ || size_ < max_size_;
        });

        if (abort_) return false;

        AVFrame* new_frame = av_frame_alloc();
        av_frame_move_ref(new_frame, frame);
        queue_.push(new_frame);
        size_++;

        cond_pop_.notify_one();
        return true;
    }

    // 取出一帧
    bool pop(AVFrame* frame) {
        std::unique_lock<std::mutex> lock(mutex_);

        cond_pop_.wait(lock, [this] {
            return abort_ || !queue_.empty();
        });

        if (abort_ && queue_.empty()) return false;

        AVFrame* front = queue_.front();
        queue_.pop();
        av_frame_move_ref(frame, front);
        av_frame_free(&front);
        size_--;

        cond_push_.notify_one();
        return true;
    }

    // 查看队首帧（不取出，用于查看 PTS 等）
    bool peek(AVFrame** frame_ptr) {
        std::lock_guard<std::mutex> lock(mutex_);
        if (queue_.empty()) return false;
        *frame_ptr = queue_.front();
        return true;
    }

    void flush() {
        std::lock_guard<std::mutex> lock(mutex_);
        while (!queue_.empty()) {
            AVFrame* f = queue_.front();
            queue_.pop();
            av_frame_free(&f);
        }
        size_ = 0;
        cond_push_.notify_all();
    }

    void abort() {
        std::lock_guard<std::mutex> lock(mutex_);
        abort_ = true;
        cond_pop_.notify_all();
        cond_push_.notify_all();
    }

    void start() {
        std::lock_guard<std::mutex> lock(mutex_);
        abort_ = false;
    }

    int size() const {
        std::lock_guard<std::mutex> lock(mutex_);
        return size_;
    }

    void set_max_size(int max) {
        std::lock_guard<std::mutex> lock(mutex_);
        max_size_ = max;
        cond_push_.notify_all();
    }

private:
    std::queue<AVFrame*> queue_;
    mutable std::mutex mutex_;
    std::condition_variable cond_pop_;
    std::condition_variable cond_push_;
    int size_ = 0;
    int max_size_ = 16;    // 帧较大，限制少一些
    bool abort_ = false;
};
```

## 12.5 Demo：测试线程安全队列

```cpp
// chapter-12-queue/main.cpp

#include "packet_queue.h"
#include "frame_queue.h"

extern "C" {
#include <libavformat/avformat.h>
#include <libavcodec/avcodec.h>
}

#include <iostream>
#include <thread>
#include <chrono>
#include <atomic>

int main(int argc, char* argv[]) {
    if (argc < 2) {
        std::cerr << "用法: " << argv[0] << " <输入文件>" << std::endl;
        return 1;
    }

    const char* input_file = argv[1];

    // 打开文件
    AVFormatContext* fmt_ctx = nullptr;
    avformat_open_input(&fmt_ctx, input_file, nullptr, nullptr);
    avformat_find_stream_info(fmt_ctx, nullptr);

    int video_idx = av_find_best_stream(fmt_ctx, AVMEDIA_TYPE_VIDEO, -1, -1, nullptr, 0);
    int audio_idx = av_find_best_stream(fmt_ctx, AVMEDIA_TYPE_AUDIO, -1, -1, nullptr, 0);

    std::cout << "视频流: #" << video_idx << std::endl;
    std::cout << "音频流: #" << audio_idx << std::endl;

    // 创建队列
    PacketQueue video_pkt_queue;
    PacketQueue audio_pkt_queue;
    video_pkt_queue.set_max_size(64);
    audio_pkt_queue.set_max_size(64);

    std::atomic<bool> finished(false);
    std::atomic<int> video_pkt_count(0);
    std::atomic<int> audio_pkt_count(0);

    // 生产者线程：解封装
    std::thread producer([&]() {
        AVPacket* pkt = av_packet_alloc();
        while (av_read_frame(fmt_ctx, pkt) >= 0) {
            if (pkt->stream_index == video_idx) {
                if (!video_pkt_queue.push(pkt)) break;
                video_pkt_count++;
            } else if (pkt->stream_index == audio_idx) {
                if (!audio_pkt_queue.push(pkt)) break;
                audio_pkt_count++;
            }
            av_packet_unref(pkt);
        }
        av_packet_free(&pkt);
        finished = true;
        video_pkt_queue.abort();
        audio_pkt_queue.abort();
        std::cout << "[生产者] 读取完成" << std::endl;
    });

    // 消费者线程 1：视频包消费
    std::thread video_consumer([&]() {
        AVPacket* pkt = av_packet_alloc();
        int count = 0;
        while (video_pkt_queue.pop(pkt)) {
            count++;
            av_packet_unref(pkt);
        }
        av_packet_free(&pkt);
        std::cout << "[视频消费者] 处理了 " << count << " 个包" << std::endl;
    });

    // 消费者线程 2：音频包消费
    std::thread audio_consumer([&]() {
        AVPacket* pkt = av_packet_alloc();
        int count = 0;
        while (audio_pkt_queue.pop(pkt)) {
            count++;
            av_packet_unref(pkt);
        }
        av_packet_free(&pkt);
        std::cout << "[音频消费者] 处理了 " << count << " 个包" << std::endl;
    });

    // 监控线程
    std::thread monitor([&]() {
        while (!finished) {
            std::cout << "  视频队列: " << video_pkt_queue.size()
                      << " | 音频队列: " << audio_pkt_queue.size()
                      << " | 已读: V=" << video_pkt_count
                      << " A=" << audio_pkt_count << std::endl;
            std::this_thread::sleep_for(std::chrono::milliseconds(500));
        }
    });

    producer.join();
    video_consumer.join();
    audio_consumer.join();
    monitor.join();

    std::cout << "\n总计: 视频 " << video_pkt_count
              << " 包, 音频 " << audio_pkt_count << " 包" << std::endl;

    avformat_close_input(&fmt_ctx);
    return 0;
}
```

## 12.6 架构总结

| 组件 | 线程 | 职责 |
| --- | --- | --- |
| Demux | 解封装线程 | 读取 AVPacket，分发到 video/audio PacketQueue |
| Video Decoder | 视频解码线程 | 从 video PacketQueue 取包，解码放入 video FrameQueue |
| Audio Decoder | 音频解码线程 | 从 audio PacketQueue 取包，解码放入 audio FrameQueue |
| Video Render | 主线程 | 从 video FrameQueue 取帧，通过 SDL2 渲染 |
| Audio Play | SDL 回调线程 | 从 audio FrameQueue 取帧，输出音频 |
| PacketQueue | 共享数据 | 线程安全的 AVPacket 缓冲队列 |
| FrameQueue | 共享数据 | 线程安全的 AVFrame 缓冲队列 |

下一章我们将基于这个架构实现多线程解封装和解码。
